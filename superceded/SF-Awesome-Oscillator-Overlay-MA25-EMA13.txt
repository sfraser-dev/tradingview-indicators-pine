// Krill, better MAs for AO
// "one day I need to find time and work with CBS on a custom AO,
// because the MAs used by the default AO kinda suck - maybe something
// with an EMA13 / MA25 duo instead of MA5/MA34 used by the default AO"
study(title="AO MAEMA", shorttitle="AO MAEMA", overlay=true)

// slow MA 
SourceMA = input(defval = close)
LengthMA = input(defval = 25, minval = 1)

// fast EMA
SourceEMA = input(defval = close)
LengthEMA = input(defval = 13, minval = 1)


// declare MA/EMAs
MA = sma(SourceMA, LengthMA)
EMA = ema(SourceEMA, LengthEMA)

// draw lines

plot(series=MA, title="", color=blue, linewidth=2,transp=0)
plot(series=EMA, title="", color=yellow, linewidth=2,transp=0)
